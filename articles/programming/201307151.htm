初始化脚本
_$.select.init($("#select1"), $("#forselect1"), {
  "muti": true
});
option参数
名称 类型 默认值 描述
editAble bool "true" 是否允许键入自定义值
muti bool "true" 是否允许选择多个
onchange function $.noop 回调函数
optChar string "," 分隔符
clearAll bool "true" 是否显示取消全选
selectAll bool "ture" 是否显示全选
外部调用接口 代码 描述
init 
_$.select.init(input, select, option);
//例
_$.select.init($("#select1"), $("#forselect1"), {
  "muti": true
});
 初始化
destory 
_$.select.destory(input);
//例
_$.select.destory($("#select1"));
 销毁
setValue 
_$.select.setValue(obj, values);
//例
_$.select.setValue($("#select1"),"SHA,PEK,Maki!");
 设值
add 
_$.select.show(obj);
//例
_$.select.show($('#select1'));
//例   PS:鼠标触发的话，需要放入队列：如下
setTimeout(function(){_$.select.show($('#select1'));});
